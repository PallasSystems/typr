15  public final class FieldDefinitionDomainFactory {
16  
17      /**
18       * Used to convert the incoming object into a DTO for egress.
19       * @param definition The definition to convert into a JACKON object.
20       * @return Null if the supplied object doesn't map to a DTO type.
21       */
22      public static SingleValidationRuleFieldDefinitionDomain getFieldDefinitionDomain(final FieldDefinition definition) {
23          final SingleValidationRuleFieldDefinitionDomain result;
24  
25          if (definition instanceof FieldDefinition) {
26              result = new SingleValidationRuleFieldDefinitionDomain((FieldDefinition)definition);
27          } else {
28              result = null;
29          }
30  
31          return result;
32      }
33  
34      /**
35       * Used to convert the incoming object into a DTO for egress.
36       * @param definition The definition to convert into a JACKON object.
37       * @return Null if the supplied object doesn't map to a DTO type.
38       */
39      public static CategoryDomain getCategoryDomain(final Category definition) {
40          final CategoryDomain result;
41  
42          if (definition instanceof Category) {
43              result = new CategoryDomain((Category)definition);
44          } else {
45              result = null;
46          }
47  
48          return result;
49      }
50  
51      /**
52       * Used to convert the incoming object into a DTO for egress.
53       * @param definition The definition to convert into a JACKON object.
54       * @return Null if the supplied object doesn't map to a DTO type.
55       */
56      public static AbstractValidationRuleDomain getValidationRuleDomain(final ValidationRule definition) {
57          final AbstractValidationRuleDomain result;
58  
59          if (definition instanceof EnumValidationRule) {
60              result = new EnumValidationRuleDomain((EnumValidationRule)definition);
61          } else if (definition instanceof EnumValidationRule) {
62              result = new EnumValidationRuleDomain((EnumValidationRule)definition);
63          } else if (definition instanceof StringValidationRule) {
64              result = new StringValidationRuleDomain((StringValidationRule)definition);
65          } else if (definition instanceof DoubleValidationRule) {
66              result = new DoubleValidationRuleDomain((DoubleValidationRule)definition);
67          } else if (definition instanceof LongValidationRule) {
68              result = new LongValidationRuleDomain((LongValidationRule)definition);
69          } else {
70              result = null;
71          }
72  
73          return result;
74      }
75  
76      /** Unusued utility class constructor. */
77      private FieldDefinitionDomainFactory() {
78          super();
79      }
80  }
