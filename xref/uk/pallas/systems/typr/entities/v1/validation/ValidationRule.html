3   /**
4    * A field definition can have multiple validation rules applied to it, this provides an interface
5    * to allow those to be managed via composition.
6    */
7   public interface ValidationRule {
8   
9     /**
10     * Retrieves a hopefully detailed description of the field definition so we can understand what it is for and
11     * why it exists.
12     *
13     * @return a hopefull long valid string (null is possible).
14     */
15    String getDescription();
16  
17    /**
18     * Sets the description to attach to this field definition.
19     *
20     * @param detailedDescription the description to attached (null is ok)
21     */
22    void setDescription(String detailedDescription);
23  
24    /**
25     * Is the supplied test object something that matches against our field definition regular expression?
26     * <p>
27     * If validation optional is set to true this will return true, if the supplied object is null, this will always
28     * return false. Other-wise this will call toString and then match the field definition regex tp confirm the object
29     * matches our desired value.
30     *
31     * @param toTest to test is valid
32     * @return false if the object fails the validation match.
33     */
34    boolean isValid(Object toTest);
35  }
