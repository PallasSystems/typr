5   import java.util.Collection;

7   /**
8    * The idea of a rule wrapper is to allow us to hold different rules for a field definition, for example postal code
9    * is different within each country (e.g. UK Post Code, USA Zip Code, etc..). Similary some things are a construction
10   * of other fields (e.g. Hostname will often have a Top Level Domain name included).
11   *
12   * The wrapper is a base interface for these different combinations to extend.
13   */
14  public interface MultiValidationRuleFieldDefinition extends FieldDefinition {
15  
16      /**
17       * Retrieves a list of all rules which have confirmed the incoming object is valid.
18       * @param toTest the object to test against the rules stored within this object
19       * @return an empty list or a list of identifiers for rules.
20       */
21      Collection<String> getRulesPassed(final Object toTest);
22  
23      /**
24       * A list of rules this field definition can validate.
25       * @return non null value (if field definition is valid).
26       */
27      Collection<RuleWrapper> getRules();
28  
29      /**
30       * Sets the validation for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
31       * @param rule the rules to add to our field definition
32       */
33      void setRules(final Collection<RuleWrapper> rule);
34  
35  }
