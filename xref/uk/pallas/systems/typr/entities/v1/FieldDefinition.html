1   package uk.pallas.systems.typr.entities.v1;
2   
3   import java.util.Collection;
4   
5   /**
6    * This class defines Complex Structured Data field definitions we can assign to various structured data schema's.
7    */
8   public interface FieldDefinition {
9   
10      /**
11       * Retrieves the shortened name (e.g. Acronym) of the field definition e.g. post code, uk mobile, IPv4, etc..
12       * @return non null value (if field definition is valid).
13       */
14      String getAcronym();
15  
16      /**
17       * Sets the shortened name (e.g. Acronym) for the field definition e.g. Ipv6, MCC, etc...
18       * @param identifier the new name for the field definition value
19       */
20      void setAcronym(final String identifier);
21  
22  
23      /**
24       * List of categories assocaited with our type. These are additional ways to define a type for routing/managing a schema.
25       * @return an empty list if nothing is supplied.
26       */
27      Collection<Category> getCategories();
28  
29      /**
30       * Sets the list of categories assocaited with our type. These are additional ways to define a type for routing/managing a schema.
31       * @param values all categories associated with the type.
32       */
33      void setCategories(final Collection<Category> values);
34  
35      /**
36       * Retrieves a hopefully detailed description of the field definition so we can understand what it is for and why it exists.
37       * @return a hopefull long valid string (null is possible).
38       */
39      String getDescription();
40  
41      /**
42       * Sets the description to attach to this field definition.
43       * @param detailedDescription the description to attache (null is ok)
44       */
45      void setDescription(final String detailedDescription);
46  
47      /**
48       * Retrieves the name of the field definition e.g. post code, uk mobile, IPv4 Address, etc..
49       * @return non null value (if field definition is valid).
50       */
51      String getName();
52  
53      /**
54       * Sets the name for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
55       * @param identifier the new name for the field definition value
56       */
57      void setName(final String identifier);
58  
59      /**
60       * Is the supplied test object something that matches against our field definition regular expression?
61       *
62       * If validation optional is set to true this will return true, if the supplied object is null, this will always
63       * return false. Otherwise this will call toString and then match the field definition regex tp confirm the object
64       * matches our desired value.
65       *
66       * @param toTest to test is valid
67       * @return false if the object fails the validation match.
68       */
69      boolean isValid(final Object toTest);
70  }
