1   package uk.pallas.systems.typr.entities.v1;
2   
3   import java.util.Collection;
4   import uk.pallas.systems.typr.entities.v1.validation.ValidationRule;
5   
6   /**
7    * This class defines Complex Structured Data field definitions we can assign to various structured data schema's.
8    */
9   public interface FieldDefinition {
10  
11    /**
12     * Retrieves the shortened name (e.g. Acronym) of the field definition e.g. post code, uk mobile, IPv4, etc..
13     *
14     * @return non null value (if field definition is valid).
15     */
16    String getAcronym();
17  
18    /**
19     * Sets the shortened name (e.g. Acronym) for the field definition e.g. Ipv6, MCC, etc...
20     *
21     * @param identifier the new name for the field definition value
22     */
23    void setAcronym(String identifier);
24  
25    /**
26     * List of categories associated with our type. These are additional ways to define a
27     * type for routing/managing a schema.
28     *
29     * @return an empty list if nothing is supplied.
30     */
31    Collection<String> getCategories();
32  
33    /**
34     * Sets the list of categories associated with our type. These are additional ways to
35     * define a type for routing/managing a schema.
36     *
37     * @param values all categories associated with the type.
38     */
39    void setCategories(Collection<String> values);
40  
41    /**
42     * Retrieves a hopefully detailed description of the field definition so
43     * we can understand what it is for and why it exists.
44     *
45     * @return a hopefull long valid string (null is possible).
46     */
47    String getDescription();
48  
49    /**
50     * Sets the description to attach to this field definition.
51     *
52     * @param detailedDescription the description to attach (null is ok)
53     */
54    void setDescription(String detailedDescription);
55  
56    /**
57     * Retrieves the name of the field definition e.g. post code, uk mobile, IPv4 Address, etc..
58     *
59     * @return non null value (if field definition is valid).
60     */
61    String getName();
62  
63    /**
64     * Sets the name for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
65     *
66     * @param identifier the new name for the field definition value
67     */
68    void setName(String identifier);
69  
70    /**
71     * Is the supplied test object something that matches against our field definition regular expression?
72     * <p>
73     * If validation optional is set to true this will return true, if the supplied object is null, this will always
74     * return false. Other-wise this will call toString and then match the field definition regex tp confirm the object
75     * matches our desired value.
76     *
77     * @param toTest to test is valid
78     * @return false if the object fails the validation match.
79     */
80    boolean isValid(Object toTest);
81  
82    /**
83     * Retrieves a list of all rules which have confirmed the incoming object is valid.
84     *
85     * @param toTest the object to test against the rules stored within this object
86     * @return an empty list or a list of identifiers for rules.
87     */
88    Collection<String> getRulesPassed(Object toTest);
89  
90    /**
91     * A list of rules this field definition can validate.
92     *
93     * @return non null value (if field definition is valid).
94     */
95    Collection<ValidationRule> getRules();
96  
97    /**
98     * Sets the validation for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
99     *
100    * @param rule the rules to add to our field definition
101    */
102   void setRules(Collection<ValidationRule> rule);
103 }
