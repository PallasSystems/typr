23  @Service
24  public class UnitsServiceImpl implements UnitsService {
25    /** Static Logger for the class. */
26    private static final Log LOGGER = LogFactory.getLog(UnitsServiceImpl.class);
27  
28    /** Cache of units against their names. **/
29    private final Map<String, Unit<?>> units;
30  
31    public UnitsServiceImpl() {
32      this.units = new ConcurrentHashMap<>();
33  
34      final Unit<Frequency> rateOfTurn = NonSI.DEGREE_ANGLE.divide(Units.MINUTE).asType(Frequency.class);
35      this.units.put("Degree Angle per Minute", rateOfTurn);
36  
37      // Pull in the other units into a unified list
38      final SI siUnits = SI.getInstance();
39      if (null != siUnits) {
40        this.processUnits(siUnits.getUnits());
41      }
42  
43      final NonSI nonSIUnits = NonSI.getInstance();
44      if (null != nonSIUnits) {
45        this.processUnits(nonSIUnits.getUnits());
46      }
47  
48      final Units unitsInstance = Units.getInstance();
49      if (null != unitsInstance) {
50        this.processUnits(unitsInstance.getUnits());
51      }
52    }
53  
54    /**
55     * Designed to iterate over a collection of units and add them to the services internal cache, so we can
56     * process a single list of units rather than having to check multiple places.
57     *
58     * @param units the units to process.
59     */
60    private void processUnits(final Set<Unit<?>> values) {
61      if (null == values) {
62        LOGGER.error("processUnits - Called Measure Unit library and did not get an associated units");
63      } else {
64        for (final Unit<?> value : values) {
65          if (null != value) {
66            if (null == value.getName() || value.getName().isBlank()) {
67              this.units.put(value.toString(), value);
68            } else {
69              this.units.put(value.getName(), value);
70            }
71          }
72        }
73      }
74    }
75  
76    /**
77     * {@inheritDoc}
78     * @return a collection of types collected from the measurement library.
79     */
80    @Override
81    public Collection<String> getUnits() {
82      return this.units.keySet();
83    }
84  
85    public Unit<?> getUnit(final String unitName) {
86      Unit<?> unit = null;
87      if (null == unitName || unitName.isBlank()) {
88        LOGGER.trace("getUnits - Invalid Unit Name was supplied");
89      } else {
90        unit = this.units.get(unitName);
91      }
92  
93      return unit;
94    }
95  
96    /**
97     * {@inheritDoc}
98     *
99     * @param unitName the name to perform a lookup on
100    * @return false if the name can't be found.
101    */
102   @Override
103   public boolean isValid(final String unitName) {
104     return ValidationRuleConstants.NO_UNITS.equals(unitName) || null != this.getUnit(unitName);
105   }
106 
107   /**
108    * {@inheritDoc}
109    *
110    * @param definition the object to check
111    * @return false if the name can't be found.
112    */
113   @Override
114   public boolean isValid(final FieldDefinition definition) {
115     final boolean result;
116 
117     if (null == definition || null == definition.getRules() || definition.getRules().isEmpty()) {
118       result = false;
119     } else {
120       boolean valid = true;
121       for (final ValidationRule rule : definition.getRules()) {
122         valid = this.isValid(rule);
123 
124         // if we find an invalid rule then stop and assume the whole definition isn't valid.
125         if (!valid) {
126           break;
127         }
128       }
129 
130       result = valid;
131     }
132 
133     return result;
134   }
135 
136   /**
137    * Checks if a Numeric based rule is found and if the unit associated with it is valid.
138    * If the rule is wrapped this will extract and test the internal rule.
139    * @param rule the rule to extract the unit value from.
140    * @return this will return true if no numeric rule, or true if a numeric rule has a valid unit
141    */
142   private boolean isValid(final ValidationRule rule) {
143     final boolean valid;
144 
145     if (rule instanceof LongValidationRule) {
146       valid = this.isValid(((LongValidationRule) rule).getUnit());
147     } else if (rule instanceof DoubleValidationRule) {
148       valid = this.isValid(((DoubleValidationRule) rule).getUnit());
149     } else if (rule instanceof CountryCodeWrapper wrapper) {
150       valid = this.isValid(wrapper.getRule());
151     } else {
152       valid = true;
153     }
154 
155     return valid;
156   }
