1   package uk.pallas.systems.typr.services;
2   
3   import java.util.Collection;
4   import uk.pallas.systems.typr.entities.v1.FieldDefinition;
5   
6   /**
7    * A wrapper for the SI Measurements library, so it can be called by various spring services.
8    */
9   public interface UnitsService {
10  
11    /**
12     * Retrieves all valid units supported by Typr.
13     * @return a collection of types collected from the measurement library.
14     */
15    Collection<String> getUnits();
16  
17    /**
18     * Looks to see if the supplied name can be found within the units library (not everything has a name).
19     *
20     * @param unitName the name to perform a lookup on
21     * @return false if the name can't be found.
22     */
23    boolean isValid(String unitName);
24  
25    /**
26     * Looks to see if the supplied field definition has Number validation rules with Units defined against them.
27     * If there is a unit this will look to see if the supplied name can be found within the units library (not
28     * everything has a name).
29     *
30     * @param definition the object to check
31     * @return false if the name can't be found.
32     */
33    boolean isValid(FieldDefinition definition);
34  }
