19  /**
20   * Used to define a single Field type and a rule to be associated with it. For example a Mobile Country Code
21   * is an international type and so this could represent a MCC value.
22   */
23  @Schema(description = "Used to define a single Field type and a rule to be associated with it. For example a Mobile "
24                        + "Country Code is an international type and so this could represent a MCC value.")
25  public class SingleValidationRuleFieldDefinitionDTO extends AbstractFieldDefinitionDTO implements SingleValidationRuleFieldDefinition {
26      /** Static Logger for the class. */
27      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractFieldDefinitionDTO.class);
28  
29      /** Validation for the field definition. */
30      @Schema(description = "Validation Rule for the field definition.",
31              anyOf={ DoubleValidationRuleDTO.class, EnumValidationRuleDTO.class, LongValidationRuleDTO.class,
32                      StringValidationRuleDTO.class })
33      private ValidationRule rule;
34  
35  
36      /** Default Class Constructor. */
37      public SingleValidationRuleFieldDefinitionDTO() {
38          this(null);
39      }
40  
41      /** Copy Constructor */
42      public SingleValidationRuleFieldDefinitionDTO(final FieldDefinition data) {
43          this(null != data ? data.getAcronym() : null, null != data ? data.getCategories() : null,
44                  null != data ? data.getDescription() : null, null != data ? data.getName() : null,
45                  null);
46      }
47  
48      /** Copy Constructor */
49      public SingleValidationRuleFieldDefinitionDTO(final SingleValidationRuleFieldDefinition data) {
50          this(null != data ? data.getAcronym() : null, null != data ? data.getCategories() : null,
51                  null != data ? data.getDescription() : null, null != data ? data.getName() : null,
52                  null != data ? data.getRule() : null);
53      }
54  
55      /**
56       * Class Constructor.
57       *
58       * @param acryo The shortened name
59       * @param cats The categories associated with our object
60       * @param desc the description of the rule
61       * @param name the name for the rule
62       * @param theRulez the Validation rule to apply
63       */
64      public SingleValidationRuleFieldDefinitionDTO(final String acryo, final Collection<Category> cats, final String desc, final String name,
65                                                    final ValidationRule theRulez) {
66          super(acryo, cats, desc, name);
67  
68          this.rule = DTOFactory.getValidationRuleDTO(theRulez);
69      }
70  
71      /**
72       * Compares the supplied object to this one, it checks the supplied object is a FieldDefinition and then checks
73       * the definition applies validation, its name and description to see if they are matching objects.
74       *
75       * @param toCompare the object to compare (can be null or a child class, etc..)
76       * @return false if the name/validation and description fields in a field definition are
77       *        different (or it isn't a field definition).
78       */
79      @Override
80      public boolean equals(final Object toCompare) {
81  
82          final boolean result;
83          if (this == toCompare) {
84              result = true;
85          } else if (toCompare instanceof SingleValidationRuleFieldDefinition) {
86              final SingleValidationRuleFieldDefinition that = (SingleValidationRuleFieldDefinition) toCompare;
87              result = super.equals(toCompare) && Objects.equals(this.getRule(), that.getRule());
88          } else {
89              result = false;
90          }
91  
92          return result;
93      }
94  
95      /**
96       * Generates a Unique hashcode for the field definition class.
97       * @return a valid integer representation of this object,
98       */
99      @Override
100     public int hashCode() {
101         return super.hashCode() + Objects.hash(this.getRule());
102     }
103 
104     /**
105      * Is the supplied test object something that matches against our field definition regular expression?
106      *
107      * If validation optional is set to true this will return true, if the supplied object is null, this will always
108      * return false. Other-wise this will call toString and then match the field definition regex tp confirm the object
109      * matches our desired value.
110      *
111      * @param toTest to test is valid
112      * @return false if the object fails the validation match.
113      */
114     @Override
115     public boolean isValid(final Object toTest) {
116         final boolean result;
117 
118         if (null == this.getRule()) {
119             result = false;
120             LOGGER.warn(String.format("isValid - No Rule to test object: %s", toTest));
121         } else {
122             result = this.getRule().isValid(toTest);
123         }
124 
125         return result;
126     }
127 
128     /**
129      * Retrieves the rule of the field definition e.g. post code, uk mobile, IPv4 Address, etc..
130      * @return non null value (if field definition is valid).
131      */
132     public ValidationRule getRule() {
133         return this.rule;
134     }
135 
136     /**
137      * Sets the validation for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
138      * @param value the rules to add to our field definition
139      */
140     public void setRule(final ValidationRule value) {
141         this.rule = DTOFactory.getValidationRuleDTO(value);
142     }
