101   public FieldDefinitionDTO(final String acryo, final Collection<String> cats,
102                             final String desc, final String fieldName, final Collection<ValidationRule> rules) {
103     this.acronym = acryo;
104     this.description = desc;
105     this.name = fieldName;
106 
107     this.categories = new HashSet<>();
108     if (null != cats) {
109       this.categories.addAll(cats);
110     }
111 
112     this.rules = new HashSet<>();
113     this.setRules(rules);

116   /**
117    * Compares the supplied object to this one, it checks the supplied object is a FieldDefinition and then checks
118    * the definition applies validation, its name and description to see if they are matching objects.
119    *
120    * @param toCompare the object to compare (can be null or a child class, etc..)
121    * @return false if the name/validation and description fields in a field definition are
122    *               different (or it isn't a field definition).
123    */
124   @Override
125   public boolean equals(final Object toCompare) {
126 
127     final boolean result;
128     if (this == toCompare) {
129       result = true;
130     } else if (toCompare instanceof FieldDefinition that) {
131       result = Objects.equals(this.getName(), that.getName())
132         && Objects.equals(this.getAcronym(), that.getAcronym())
133         && Objects.equals(this.getDescription(), that.getDescription())
134         && this.getCategories().containsAll(that.getCategories())
135         && that.getCategories().containsAll(this.getCategories());
136     } else {
137       result = false;
138     }
139 
140     return result;
141   }

154   /**
155    * Retrieves a list of all rules which have confirmed the incoming object is valid.
156    *
157    * @param toTest the object to test against the rules stored within this object
158    * @return an empty list or a list of identifiers for rules.
159    */
160   public Collection<String> getRulesPassed(final Object toTest) {
161     final Collection<String> passed = new HashSet<>();
162 
163     for (final ValidationRule rule : this.getRules()) {
164       if (rule.isValid(toTest)) {
165         passed.add(rule.toString());
166       }
167     }
168 
169     return passed;
170   }

172   /**
173    * Is the supplied test object something that matches against our field definition regular expression?
174    * <p>
175    * If validation optional is set to true this will return true, if the supplied object is null, this will always
176    * return false. Other-wise this will call toString and then match the field definition regex tp confirm the object
177    * matches our desired value.
178    *
179    * @param toTest to test is valid
180    * @return false if the object fails the validation match.
181    */
182   @Override
183   public boolean isValid(final Object toTest) {
184     final Collection<String> results = this.getRulesPassed(toTest);
185     return null != results && !results.isEmpty();
186   }

277   /**
278    * A list of rules this field definition can validate.
279    *
280    * @return non null value (if field definition is valid).
281    */
282   public Collection<ValidationRule> getRules() {
283     final Collection<ValidationRule> results = new HashSet<>();
284 
285     for (final ValidationRule rule : this.rules) {
286       if (rule instanceof CountryCodeWrapper) {
287         results.add(new CountryCodeRuleWrapperDTO((CountryCodeWrapper)rule));
288       } else if (rule instanceof DoubleValidationRule) {
289         results.add(new DoubleValidationRuleDTO((DoubleValidationRule)rule));
290       } else if (rule instanceof EnumValidationRule) {
291         results.add(new EnumValidationRuleDTO((EnumValidationRule)rule));
292       } else if (rule instanceof LongValidationRule) {
293         results.add(new LongValidationRuleDTO((LongValidationRule)rule));
294       } else if (rule instanceof StringValidationRule) {
295         results.add(new StringValidationRuleDTO((StringValidationRule)rule));
296       }
297     }
298 
299     return results;
300   }

302   /**
303    * Sets the validation for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
304    *
305    * @param validationRules the rules to add to our field definition
306    */
307   public void setRules(final Collection<ValidationRule> validationRules) {
308 
309     this.rules.clear();
310 
311     if (null != validationRules) {
312       for (final ValidationRule rule : validationRules) {
313         if (rule instanceof CountryCodeWrapper) {
314           this.rules.add(new CountryCodeRuleWrapperDTO((CountryCodeWrapper)rule));
315         } else if (rule instanceof DoubleValidationRule) {
316           this.rules.add(new DoubleValidationRuleDTO((DoubleValidationRule)rule));
317         } else if (rule instanceof EnumValidationRule) {
318           this.rules.add(new EnumValidationRuleDTO((EnumValidationRule)rule));
319         } else if (rule instanceof LongValidationRule) {
320           this.rules.add(new LongValidationRuleDTO((LongValidationRule)rule));
321         } else if (rule instanceof StringValidationRule) {
322           this.rules.add(new StringValidationRuleDTO((StringValidationRule)rule));
323         }
324       }
325     }
