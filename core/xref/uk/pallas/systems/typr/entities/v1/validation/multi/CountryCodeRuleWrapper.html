6   /**
7    *  Some field definitions are country specific for example a postal code within the UK is difined differently
8    *  to a zip code in the United States but both are used within the same area of an address block
9    */
10  public interface CountryCodeRuleWrapper extends RuleWrapper {
11  
12      /**
13       * The specific country this rule is associated with.
14       * @return UNDEFINED or a valid country code.
15       */
16      CountryCode getCountryCode();
17  
18      /**
19       * Sets the specific country this rule is associated with.
20       * @param code - UNDEFINED or a valid country code.
21       */
22      void setCountryCode(final CountryCode code);
23  
24      /**
25       * Retrieves the name of the field definition e.g. post code, uk mobile, IPv4 Address, etc..
26       * @return non null value (if field definition is valid).
27       */
28      String getName();
29  
30      /**
31       * Sets the name for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
32       * @param identifier the new name for the field definition value
33       */
34      void setName(final String identifier);
35  }
