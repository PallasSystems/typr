3   /**
4    * Defines regular expression rules for a field to confirm if the field is valid or not.
5    */
6   public interface StringValidationRule extends ValidationRule {
7   
8     /**
9      * The regular Expression to apply to a string defined by this field definition, to confirm a given string is valid.
10     *
11     * @return non-null regular expression that can be used by Java Regex system.
12     */
13    String getDetectRegex();
14  
15    /**
16     * The regular Expression to apply to a string defined by this field definition, to confirm a given string is valid.
17     *
18     * @param validation a valid regular expression string.
19     */
20    void setDetectRegex(String validation);
21  
22    /**
23     * The regular Expression to extract a matching regex string from a text block.
24     *
25     * @return non null regular expression that can be used by Java Regex system.
26     */
27    String getExtractRegex();
28  
29    /**
30     * Allows us to set the regular expression to extract the term from a text block.
31     *
32     * @param validation a valid regular expression string.
33     */
34    void setExtractRegex(String validation);
35  
36    /**
37     * Is the supplied test object something that matches against our field definition regular expression?
38     * <p>
39     * If validation optional is set to true this will return true, if the supplied object is null, this will always
40     * return false. Otherwise it will call matches on the supplied string.
41     *
42     * @param toTest to test is valid
43     * @return false if the object fails the validation match.
44     */
45    boolean isValid(String toTest);
46  }
