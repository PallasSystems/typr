8    */
9   public interface FieldDefinitionService {
10  
11    /**
12     * Retrieve all field definitions from the database.
13     * @return an empty collection or valid definitions
14     */
15    Collection<FieldDefinition> getFieldDefinitions();
16  
17    /**
18     * Retrieve all field definitions which are a member of the supplied category from the database.
19     * @return an empty collection or valid definitions
20     */
21    Collection<FieldDefinition> getFieldDefinitionsByCategory(String string);
22  
23    /**
24     * Retrieve a speciic field definition through its name
25     * @param name the name (primary key) of the field definition
26     * @return null if the field definition can not be found.
27     */
28    FieldDefinition getFieldDefinitionByName(String name);
29  
30    /**
31     * Creates a field definition within the database
32     * @param definition the definition to be created within the database
33     * @return null if there was a problem saving the definition
34     */
35    FieldDefinition saveFieldDefintion(FieldDefinition definition);
