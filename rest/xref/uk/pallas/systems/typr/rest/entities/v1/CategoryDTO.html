15  
16      /** Detailed description of the field definition. */

18      @Size(min=0, max=4096)

21      /** Name of the field definition e.g. post code, uk mobile. */
22      @Schema(description = "Name of the category, this is", nullable = false, example="Edge")

28          this(null);
29      }
30  
31      /** Class Copy Constructor. */

33          this(null == data ? null : data.getName(), null == data ? null : data.getDescription());
34      }
35  
36      /** Class Constructor which lets us set all fields within the object. */
37      public CategoryDTO(final String identifier, final String desc) {
38          super();
39  
40          this.name = identifier;
41          this.description = desc;
42      }
43  
44      /**
45       * Retrieves the name of the field definition e.g. post code, uk mobile, IPv4, etc..
46       * @return non null value (if field definition is valid).
47       */
48      public String getName() {

50      }
51  
52      /**
53       * Sets the name for the field definition e.g. Post Code, Ipv6, MCC, etc...
54       * @param identifier the new name for the field definition value
55       */
56      public void setName(final String identifier) {

58      }
59  
60      /**
61       * Retrieves a hopefully detailed description of the field definition so we can understand what it is for and why it exists.
62       * @return a hopefull long valid string (null is possible).
63       */
64      public String getDescription() {
65          return this.description;
66      }
67  
68      /**
69       * Sets the description to attach to this field definition.
70       * @param detailedDescription the description to attache (null is ok)
71       */
72      public void setDescription(final String detailedDescription) {
73          this.description = detailedDescription;
74      }
