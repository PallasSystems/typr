110   /**
111    * Compares the supplied object to this one, it checks the supplied object is a FieldDefinition and then checks
112    * the definition applies validation, its name and description to see if they are matching objects.
113    *
114    * @param toCompare the object to compare (can be null or a child class, etc..)
115    * @return false if the name/validation and description fields in a field definition are
116    *               different (or it isn't a field definition).
117    */
118   @Override
119   public boolean equals(final Object toCompare) {
120 
121     final boolean result;
122     if (this == toCompare) {
123       result = true;
124     } else if (toCompare instanceof FieldDefinition that) {
125       result = Objects.equals(this.getName(), that.getName())
126         && Objects.equals(this.getAcronym(), that.getAcronym())
127         && Objects.equals(this.getDescription(), that.getDescription())
128         && this.getCategories().containsAll(that.getCategories())
129         && that.getCategories().containsAll(this.getCategories());
130     } else {
131       result = false;
132     }
133 
134     return result;
135   }

148   /**
149    * Retrieves a list of all rules which have confirmed the incoming object is valid.
150    *
151    * @param toTest the object to test against the rules stored within this object
152    * @return an empty list or a list of identifiers for rules.
153    */
154   public Collection<String> getRulesPassed(final Object toTest) {
155     final Collection<String> passed = new HashSet<>();
156 
157     for (final ValidationRule rule : this.getRules()) {
158       if (rule.isValid(toTest)) {
159         passed.add(rule.toString());
160       }
161     }
162 
163     return passed;
164   }

166   /**
167    * Is the supplied test object something that matches against our field definition regular expression?
168    * <p>
169    * If validation optional is set to true this will return true, if the supplied object is null, this will always
170    * return false. Other-wise this will call toString and then match the field definition regex tp confirm the object
171    * matches our desired value.
172    *
173    * @param toTest to test is valid
174    * @return false if the object fails the validation match.
175    */
176   @Override
177   public boolean isValid(final Object toTest) {
178     final Collection<String> results = this.getRulesPassed(toTest);
179     return null != results && !results.isEmpty();
180   }

271   /**
272    * A list of rules this field definition can validate.
273    *
274    * @return non null value (if field definition is valid).
275    */
276   public Collection<ValidationRule> getRules() {
277     final Collection<ValidationRule> results = new HashSet<>();
278 
279     for (final ValidationRule rule : this.rules) {
280       if (rule instanceof CountryCodeWrapper) {
281         results.add(new CountryCodeRuleWrapperDTO((CountryCodeWrapper)rule));
282       } else if (rule instanceof DoubleValidationRule) {
283         results.add(new DoubleValidationRuleDTO((DoubleValidationRule)rule));
284       } else if (rule instanceof EnumValidationRule) {
285         results.add(new EnumValidationRuleDTO((EnumValidationRule)rule));
286       } else if (rule instanceof LongValidationRule) {
287         results.add(new LongValidationRuleDTO((LongValidationRule)rule));
288       } else if (rule instanceof StringValidationRule) {
289         results.add(new StringValidationRuleDTO((StringValidationRule)rule));
290       }
291     }
292 
293     return results;
294   }

296   /**
297    * Sets the validation for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
298    *
299    * @param validationRules the rules to add to our field definition
300    */
301   public void setRules(final Collection<ValidationRule> validationRules) {
302 
303     this.rules.clear();
304 
305     if (null != validationRules) {
306       for (final ValidationRule rule : validationRules) {
307         if (rule instanceof CountryCodeWrapper) {
308           this.rules.add(new CountryCodeRuleWrapperDTO((CountryCodeWrapper)rule));
309         } else if (rule instanceof DoubleValidationRule) {
310           this.rules.add(new DoubleValidationRuleDTO((DoubleValidationRule)rule));
311         } else if (rule instanceof EnumValidationRule) {
312           this.rules.add(new EnumValidationRuleDTO((EnumValidationRule)rule));
313         } else if (rule instanceof LongValidationRule) {
314           this.rules.add(new LongValidationRuleDTO((LongValidationRule)rule));
315         } else if (rule instanceof StringValidationRule) {
316           this.rules.add(new StringValidationRuleDTO((StringValidationRule)rule));
317         }
318       }
319     }
