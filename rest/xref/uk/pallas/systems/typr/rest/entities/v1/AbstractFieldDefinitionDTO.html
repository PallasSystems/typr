20  public abstract class AbstractFieldDefinitionDTO implements FieldDefinition {
21      /** Static Logger for the class. */
22      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractFieldDefinitionDTO.class);
23  
24      /** The shortened name (e.g. Acronym) of the field definition e.g. post code, uk mobile, IPv4, etc.. */
25      @Size(min=0, max=4096)
26      @Schema(description = "The shortened name (e.g. Acronym) of the field definition e.g. post code, uk mobile, IPv4, etc..",
27              example="MCC")
28      private String acronym;
29  
30      /** List of categories associated with our type. */
31      @ArraySchema(schema = @Schema(description = "List of categories associated with our type.",
32              implementation=CategoryDTO.class),
33              uniqueItems = true,
34              minItems=0)
35      private Collection<Category> categories;
36  
37      /** Detailed description of the field definition. */
38      @Schema(description = "Detailed description of the field definition.",
39              example="The mobile country code consists of three decimal digits and the mobile network code consists of two or three decimal digits ")
40      @Size(min=0, max=4096)
41      private String description;
42  
43      /** Name of the field definition e.g. post code, uk mobile. */
44      @Schema(description = "Name of the field definition e.g. post code, uk mobile.",
45              example = "Mobile Country Code",
46              nullable = false)
47      @Size(min=1, max=100)
48      private String name;
49  
50      /** Default Class Constructor. */
51      public AbstractFieldDefinitionDTO() {
52          this(null);
53      }
54  
55      /** Copy Constructor */
56      public AbstractFieldDefinitionDTO(final FieldDefinition data) {
57          this(null != data ? data.getAcronym() : null, null != data ? data.getCategories() : null,
58                  null != data ? data.getDescription() : null, null != data ? data.getName() : null);
59      }
60  
61      /**
62       * Class Constructor.
63       *
64       * @param acryo The shortened name
65       * @param cats The categories associated with our object
66       * @param desc the description of the rule
67       * @param name the name for the rule
68       */
69      public AbstractFieldDefinitionDTO(final String acryo, final Collection<Category> cats,
70                                        final String desc, final String name) {
71          super();
72  
73          this.acronym = acryo;
74          this.description = desc;
75          this.name = name;
76  
77          this.categories = new HashSet<>();
78          if (null != cats) {
79              this.categories.addAll(cats.stream().filter(value -> null != value)
80                                                  .map(value -> new CategoryDTO(value))
81                                                  .collect(Collectors.toSet()));
82          }
83      }
84  
85      /**
86       * Compares the supplied object to this one, it checks the supplied object is a FieldDefinition and then checks
87       * the definition applies validation, its name and description to see if they are matching objects.
88       *
89       * @param toCompare the object to compare (can be null or a child class, etc..)
90       * @return false if the name/validation and description fields in a field definition are
91       *        different (or it isn't a field definition).
92       */
93      @Override
94      public boolean equals(final Object toCompare) {
95  
96          final boolean result;
97          if (this == toCompare) {
98              result = true;
99          } else if (toCompare instanceof FieldDefinition) {
100             final FieldDefinition that = (FieldDefinition) toCompare;
101             result = Objects.equals(this.getName(), that.getName())
102                     && Objects.equals(this.getAcronym(), that.getAcronym())
103                     && Objects.equals(this.getDescription(), that.getDescription())
104                     && (this.getCategories().containsAll(that.getCategories())
105                     && that.getCategories().containsAll(this.getCategories()));
106         } else {
107             result = false;
108         }
109 
110         return result;
111     }
112 
113     /**
114      * Generates a Unique hashcode for the field definition class.
115      * @return a valid integer representation of this object,
116      */
117     @Override
118     public int hashCode() {
119         return Objects.hash(this.getAcronym(), this.getName(), this.getDescription());
120     }

122     /**
123      * Retrieves the shortened name (e.g. Acronym) of the field definition e.g. post code, uk mobile, IPv4, etc..
124      * @return non null value (if field definition is valid).
125      */
126     @Override
127     public String getAcronym() {
128         return acronym;
129     }

132      * Sets the shortened name (e.g. Acronym) for the field definition e.g. Ipv6, MCC, etc...
133      * @param identifier the new name for the field definition value
134      */
135     @Override
136     public void setAcronym(final String identifier) {
137         this.acronym = identifier;
138     }

140     /**
141      * List of categories assocaited with our type. These are additional ways to define a type for routing/managing a schema.
142      * @return an empty list if nothing is supplied.
143      */
144     @Override
145     public Collection<Category> getCategories() {
146         return new HashSet<>(categories);
147     }
148 
149     /**
150      * Sets the list of categories assocaited with our type. These are additional ways to define a type for routing/managing a schema.
151      * @param values all categories associated with the type.
152      */
153     @Override
154     public void setCategories(final Collection<Category> values) {
155         if (null == values) {
156             this.categories.clear();
157         } else {
158             this.categories.clear();
159             this.categories.addAll(values.stream().filter(value -> null != value)
160                     .map(value -> new CategoryDTO(value))
161                     .collect(Collectors.toSet()));
162         }
163     }
164 
165     /**
166      * Retrieves a hopefully detailed description of the field definition so we can understand what it is for and why it exists.
167      * @return a hopefull long valid string (null is possible).
168      */
169     @Override
170     public String getDescription() {
171         return description;
172     }
173 
174     /**
175      * Sets the description to attach to this field definition.
176      * @param detailedDescription the description to attache (null is ok)
177      */
178     @Override
179     public void setDescription(final String detailedDescription) {
180         this.description = detailedDescription;
181     }
182 
183     /**
184      * Retrieves the name of the field definition e.g. post code, uk mobile, IPv4 Address, etc..
185      * @return non null value (if field definition is valid).
186      */
187     @Override
188     public String getName() {
189         return this.name;
190     }
191 
192     /**
193      * Sets the name for the field definition e.g. Post Code, Ipv6, Mobile Country Code, etc...
194      * @param identifier the new name for the field definition value
195      */
196     @Override
197     public void setName(final String identifier) {
198         this.name = identifier;
199     }
