1   package uk.pallas.systems.typr.rest;
2   
3   import org.springframework.beans.factory.annotation.Autowired;
4   import org.springframework.http.HttpStatus;
5   import org.springframework.web.bind.annotation.*;
6   import org.springframework.web.server.ResponseStatusException;
7   import uk.pallas.systems.typr.entities.v1.*;
8   import uk.pallas.systems.typr.entities.v1.impl.DoubleFieldDefinitionDTO;
9   import uk.pallas.systems.typr.entities.v1.impl.LongFieldDefinitionDTO;
10  import uk.pallas.systems.typr.entities.v1.impl.StringFieldDefinitionDTO;
11  import uk.pallas.systems.typr.services.FieldDefinitionServices;
12  
13  import java.util.Collection;
14  import java.util.stream.Collectors;
15  
16  /**
17   * This Defines an interface for retrieving Field Definitions stored within Typr.
18   */
19  @RestController("Typr API")
20  @RequestMapping("api/type/v1")
21  public class FieldDefinitionsController {
22  
23    /** The backend service to retrieve. */
24    @Autowired
25    private FieldDefinitionServices services;
26  
27    /**
28     * Retrieves all Field Definitions held within Typr.
29     * @return A collection of Long, Double, String, Enumerate, etc... field definitions.
30     */
31    @GetMapping("/types")
32    public Collection<FieldDefinition> getTypes() {
33  
34      final Collection<FieldDefinition> definitions = this.services.getFieldDefinitions();
35      if (null == definitions || definitions.isEmpty()) {
36        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "No definitions stored within Typr" );
37      }
38  
39      return definitions.stream().filter(result -> null != result)
40          .map(result -> this.getDTO(result))
41          .collect(Collectors.toList());
42    }
43  
44    /**
45     * Retrieves all Field Definitions of a specific type within the system.
46     * @param type The Type of Field Definitions you wish to retrieve (e.g. string, enum, double, long, number).
47     * @return A collection of definitions of a specific type.
48     */
49    @GetMapping("/types/type/{type}")
50    public Collection<FieldDefinition> getFieldDefinitionByType(@PathVariable(name="type", required = true) final String type) {
51  
52      final Collection<FieldDefinition> definitions = this.services.getFieldDefinitionsByType(type);
53      if (null == definitions || definitions.isEmpty()) {
54        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "No definitions for type:" + type);
55      }
56  
57      return definitions.stream().filter(result -> null != result)
58                                 .map(result -> this.getDTO(result))
59                                 .collect(Collectors.toList());
60    }
61  
62    /**
63     * Retrieves a Field Definition based on the full length name.
64     * @param name The name of the field definition we want to retrieve from Typr.
65     * @return A valid Field definition object if one is found within the database.
66     */
67    @GetMapping("/type/name/{name}")
68    public FieldDefinition getFieldDefinitionByName(@PathVariable(name="name", required = true) final String name) {
69      final FieldDefinition result = this.getDTO(this.services.getFieldDefinitionByName(name));
70  
71      if (null == result) {
72        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "No type definition found for:" + name);
73      }
74  
75      return result;
76    }
77  
78    /**
79     * Used to insert new Field Definitions into the Typr system.
80     * @param definition The definition to insert into the system.
81     * @return The stored version of the field definition.
82     */
83    @PutMapping("/type")
84    public FieldDefinition putType(@RequestBody(required = true) final FieldDefinition definition) {
85      final FieldDefinition saved = this.services.saveFieldDefintion(definition);
86      return this.getDTO(saved);
87    }
88  
89    /**
90     * Used to convert the incoming object into a DTO for egress.
91     * @param definition The definition to convert into a JACKON object.
92     * @return Null if the supplied object doesn't map to a DTO type.
93     */
94    private FieldDefinition getDTO(final FieldDefinition definition) {
95      final FieldDefinition result;
96  
97      if (definition instanceof StringFieldDefinition) {
98        result = new StringFieldDefinitionDTO((StringFieldDefinition)definition);
99      } else if (definition instanceof DoubleFieldDefinition) {
100       result = new DoubleFieldDefinitionDTO((DoubleFieldDefinition)definition);
101     } else if (definition instanceof LongFieldDefinition) {
102       result = new LongFieldDefinitionDTO((LongFieldDefinition)definition);
103     } else if (definition instanceof NumberFieldDefinition) {
104       result = new DoubleFieldDefinitionDTO((NumberFieldDefinition)definition);
105     } else {
106       result = null;
107     }
108 
109     return result;
110   }
111 }
