20  public class DomainFactory {
21      /** Static Logger for the class. */
22      private static final Logger LOGGER = LoggerFactory.getLogger(DomainFactory.class);
23  
24      public static AbstractFieldDefinitionDomain getFieldDefinitionDTO(final FieldDefinition value) {
25          final AbstractFieldDefinitionDomain result;
26  
27          if (value instanceof SingleValidationRuleFieldDefinition) {
28              result = new SingleValidationRuleFieldDefinitionDomain((SingleValidationRuleFieldDefinition)value);
29          } else {
30              result = null;
31              LOGGER.warn(String.format("getFieldDefinitionDTO - Unsupported Definition supplied: %s", value));
32          }
33  
34          return result;
35      }
36  
37  
38      public static AbstractValidationRuleDomain getValidationRuleDomain(final ValidationRule value) {
39          final AbstractValidationRuleDomain result;
40  
41          if (value instanceof DoubleValidationRule) {
42              result = new DoubleValidationRuleDomain((DoubleValidationRule)value);
43          } else if (value instanceof EnumValidationRule) {
44              result = new EnumValidationRuleDomain((EnumValidationRule)value);
45          } else if (value instanceof LongValidationRule) {
46              result = new LongValidationRuleDomain((LongValidationRule)value);
47          } else if (value instanceof StringValidationRule) {
48              result = new StringValidationRuleDomain((StringValidationRule)value);
49          } else {
50              result = null;
51              LOGGER.warn(String.format("getValidationRuleDomain - Unsupported Rule Type supplied: %s", value));
52          }
53  
54          return result;
55      }
56  
57      /** Unusued utility class constructor. */
58      private DomainFactory() {
59          super();
60      }
61  }
