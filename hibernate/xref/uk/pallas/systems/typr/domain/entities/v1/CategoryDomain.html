1   package uk.pallas.systems.typr.domain.entities.v1;
2   
3   import jakarta.persistence.*;
4   import uk.pallas.systems.typr.entities.v1.Category;
5   
6   /**
7    * This class defines Categories which can be associated with a definition.
8    */
9   @Entity
10  public class CategoryDomain implements Category {
11  
12      /** Detailed description of the field definition. */
13      @Column(length=4096, nullable = true)
14      private String description;
15  
16      /** Name of the field definition e.g. post code, uk mobile. */
17      @Id
18      @Column(length=4096, nullable = false)
19      private String name;
20  
21      /** Default Class Constructor, sets values to null. */
22      public CategoryDomain() {
23          this(null);
24      }
25  
26      /** Class Copy Constructor. */
27      public CategoryDomain(final Category data) {
28          this(null == data ? null : data.getName(), null == data ? null : data.getDescription());
29      }
30  
31      /** Class Constructor which lets us set all fields within the object. */
32      public CategoryDomain(final String identifier, final String desc) {
33          super();
34  
35          this.name = identifier;
36          this.description = desc;
37      }
38  
39      /**
40       * Retrieves the name of the field definition e.g. post code, uk mobile, IPv4, etc..
41       * @return non null value (if field definition is valid).
42       */
43      public String getName() {
44          return this.name;
45      }
46  
47      /**
48       * Sets the name for the field definition e.g. Post Code, Ipv6, MCC, etc...
49       * @param identifier the new name for the field definition value
50       */
51      public void setName(final String identifier) {
52          this.name = identifier;
53      }
54  
55      /**
56       * Retrieves a hopefully detailed description of the field definition so we can understand what it is for and why it exists.
57       * @return a hopefull long valid string (null is possible).
58       */
59      public String getDescription() {
60          return this.description;
61      }
62  
63      /**
64       * Sets the description to attach to this field definition.
65       * @param detailedDescription the description to attache (null is ok)
66       */
67      public void setDescription(final String detailedDescription) {
68          this.description = detailedDescription;
69      }
70  }
