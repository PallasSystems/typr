4   import uk.pallas.systems.typr.entities.v1.validation.number.DoubleValidationRule;

6   import jakarta.persistence.Entity;
7   
8   @Entity
9   public class DoubleValidationRuleDomain extends AbstractNumberValidationRuleDomain<Double> implements DoubleValidationRule {
10  
11      /**
12       * Default constructor, sets everything to null and makes validation optional.
13       */
14      public DoubleValidationRuleDomain() {
15          this(null);
16      }
17  
18      /**
19       * Copy Constructor, passes up to parent Copy Constructor and sets Minimum/Maximum values.
20       * @param data the object to create a duplicate off.
21       */
22      public DoubleValidationRuleDomain(final DoubleValidationRule data) {
23          this(null == data ? null : data.getMaximumValue(), null == data ? null : data.getMinimumValue(),
24                  null == data ? null : data.getDescription());
25      }
26  
27      /**
28       * Constructor, allows us to set the internal abstract fields
29       *
30       * @param max                The upper bound allowed for the field
31       * @param min                the lower bound allowed for the field
32       */
33      protected DoubleValidationRuleDomain(final Double max, final Double min, final String detailedDescription) {
34          super(max, min, detailedDescription);
35      }
36  
37      /**
38       * Attempts to confirm the supplied value is a whole number and will return it is true.
39       *
40       * @param toConvert the object to test (if Number or a String holding a valid whole number.
41       * @return null .. or a valid Number.
42       */
43      @Override
44      protected Double getNumber(final Object toConvert) {
45  
46          Double result = null;
47  
48          if (toConvert instanceof Number) {
49              result = ((Number) toConvert).doubleValue();
50          } else if (null != toConvert) {
51              try {
52                  result = Double.parseDouble(toConvert.toString());
53              } catch (final NumberFormatException exception) {
54                  // TODO lets catch this fail and log it.
55              }
56          }
57  
58          return result;
59      }
60  }
