12    implements DoubleValidationRule {
13  
14    /** Static Logger for the class. */
15    private static final Log LOGGER = LogFactory.getLog(DoubleValidationRuleDomain.class);
16  
17    /**
18     * Default constructor, sets everything to null and makes validation optional.
19     */
20    public DoubleValidationRuleDomain() {
21      this(null);
22    }
23  
24    /**
25     * Copy Constructor, passes up to parent Copy Constructor and sets Minimum/Maximum values.
26     *
27     * @param data the object to create a duplicate off.
28     */
29    public DoubleValidationRuleDomain(final DoubleValidationRule data) {
30      super(data);
31    }
32  
33    /**
34     * Constructor, allows us to set the internal abstract fields.
35     *
36     * @param max The upper bound allowed for the field
37     * @param min the lower bound allowed for the field
38     * @param detailedDescription Detailed description of the field definition.
39     * @param unitName Defines a Unit name, to be used to when the same field has different units.
40     */
41    public DoubleValidationRuleDomain(final Double max, final Double min, final String detailedDescription,
42                                      final String unitName) {
43      super(max, min, detailedDescription, unitName);
44    }
45  
46    /**
47     * Attempts to confirm the supplied value is a whole number and will return it is true.
48     *
49     * @param toConvert the object to test (if Number or a String holding a valid whole number.
50     * @return null .. or a valid Number.
51     */
52    @Override
53    protected Double getNumber(final Object toConvert) {
54  
55      Double result = null;
56  
57      if (toConvert instanceof Number) {
58        result = ((Number) toConvert).doubleValue();
59      } else if (null != toConvert) {
60        try {
61          result = Double.parseDouble(toConvert.toString());
62        } catch (final NumberFormatException exception) {
63          if (LOGGER.isInfoEnabled()) {
64            LOGGER.info(String.format("getNumber - Unable to convert %s to Double", toConvert));
65          }
66        }
67      }
68  
69      return result;
70    }
71  }
