3   import org.junit.jupiter.api.Assertions;

8   class DoubleValidationRuleDTOTest
9     extends AbstractNumberValidationRuleDTOTest<DoubleValidationRuleDTO, Double> {
10  
11    /** Used to ensure the valid number is within our defined range. by taking the current time and removing 1 minute). */
12    private static final double MILLS_IN_MINUTE = 60000;
13  
14    @Override
15    DoubleValidationRuleDTO generateTestInstance() {
16      final double max = MILLS_IN_MINUTE * 2;
17      final double min = 10.0;
18      final String description = "Test description";
19      final String unitName = "Knot";
20  
21      return new DoubleValidationRuleDTO(max,min,description, unitName);
22    }
23  
24    /**
25     * The Max range in auto generated is the current time, so this removes 1 minute from the
26     * current time to ensure the number is lower.
27     * @return a valid double within the generateTestInstance range.
28     */
29    @Override
30    Double getValidNumber() {
31      return MILLS_IN_MINUTE;
32    }
33  
34    @Test
35    void testConstructor() {
36  
37      final double max = 1000.0;
38      final double min = 100.0;
39      final String description = "DoubleValidationRuleDTOTest-testConstructor";
40      final String unitName = ValidationRuleConstants.NO_UNITS;
41  
42      final DoubleValidationRule basic = new DoubleValidationRuleDTO(max,min,description, unitName);
43      Assertions.assertEquals(max, basic.getMaximumValue());
44      Assertions.assertEquals(min, basic.getMinimumValue());
45      Assertions.assertEquals(description, basic.getDescription());
46      Assertions.assertEquals(unitName, basic.getUnit());
47    }
48  
49    @Test
50    void testCopyConstructor() {
51      final DoubleValidationRule basic = this.generateTestInstance();
52      final DoubleValidationRule copy = new DoubleValidationRuleDTO(basic);
53  
54      Assertions.assertEquals(basic, copy);
55    }
